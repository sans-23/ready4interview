        while ((pos = result.find("teh")) != string::npos) {
            result.replace(pos, 3, "the");
        }
        while ((pos = result.find("adn")) != string::npos) {

        string word;
        int count = 0;

        }
        return text + "\n[Word count: " + to_string(count) + "]";
    }

    string getName() override {
        return "Word Count";
    }
};
